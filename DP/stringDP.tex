Edit distance $S_1$ 最少需要經過幾次增、刪或換字變成 $S_2$
\begin{tiny}
    \[
        dp[i][j] = 
        \begin{cases}
            \begin{array}{cl}
                i+1, & \mbox{if} \ j=-1 \\
                j+1, & \mbox{if} \ i=-1 \\
                dp[i-1][j-1], & \mbox{if} \ S_1[i] = S_2[j] \\
                \min\left \{
                    \begin{array}{c}
                        dp[i][j-1] \\ dp[i-1][j] \\ dp[i-1][j-1]
                    \end{array}\right
                \}+1, & \mbox{if} \ S_1[i] \neq S_2[j]
            \end{array}
        \end{cases}
    \]
\end{tiny}

\begin{tiny}
    \[
        dp[l][r] =
        \begin{cases}
            \begin{array}{crc}
                1 & \mbox{if} & l = r \\
                dp[l+1][r-1] & \mbox{if} & S[l] = S[r] \\
                \max\{dp[l+1][r], dp[l][r-1]\} & \mbox{if} & S[l] \neq S[r] \\
            \end{array}
        \end{cases}
    \]
\end{tiny}
